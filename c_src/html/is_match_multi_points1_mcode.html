1function [hits] = is_match_multi_points(mono, Fs, keys, threshold)
 2
 3%some comments are required here
 4% mono      - the single mono track audio
 5% Fs        - the sampling rate of the track
 6% keys      - a simple integer data structure representing keys. Format is
 7% as follows: 
 8%  
 9% [f1,f2,f3, 0, g1,g2,g3, -1.....] -- where f and g are different keys
10% 
11% Note that the number of points must be symetric. 
12%
13% threshold - the minimum spectral density of a match
14
15
16
17NFFT = 16384;
18% 2^nextpow2(length(mono)); %
19Y = fft(mono,NFFT);
20transform = Y.*conj(Y)/NFFT;
21f = Fs/2*linspace(0,1,NFFT/2+1);
22
23key_points = find(keys==0);
24key_start = 1;
25max_hits = 0;
26
27% for each key, loop through and extract this key.
28for i=1:length(key_points)
29    
30    % extract the key
31    key = keys(key_start:key_points(i)-1);
32    
33    key_hits = zeros(1, length(key));
34
35    % for each key, find matches.
36    for j=1:length(key)
37       
38        % fprintf('Looking at Key: %d\n', key(j));
39
40        
41        possible_matches = find(f >= (key(j)-100) & f <= (key(j)+100));
42    
43        for m=1:length(possible_matches)
44            freq = possible_matches(m);
45            
46            if transform(freq) >= threshold
47                % one more hit -- every key must be hit at least once.
48                key_hits(j) = key_hits(j)+1;
49            end
50        end
51    end
52    
53    % after looking at this entire key, if there aren't any keys that
54    % didn't hit, see if we want to up the max
55    if isempty(find(key_hits==0,1))
56        if sum(key_hits) > max_hits
57            max_hits = sum(key_hits);
58        end
59    end
60    
61    % update the key position
62    key_start = key_points(i)+1;
63    
64end
65    
66   
67hits = max_hits;
